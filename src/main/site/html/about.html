Call functions in one script, from another

library.lsl 

...
string some_function(string _input) {
	// do stuff
}
...

your_script.lsl 
default
{
	state_entry
	{
		string result = callOtherScript( "library.lsl", "some_function", "the input" );
		llOwnerSay( "The other script said '" + result + "'!" );
	}
}
